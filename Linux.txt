1.  введите список всех фалов в папке html 

	cd /var/www/html #Перейти в папку html
	ls -a #Вывести все файлы, которые лежат в папке html


2. найдите в файле  access.log все упоминания строки "404"
	cd ../../logs #Выйти из папки html и перейти в папку logs access.log
	grep -o  '404' access.log #Вывести все упоминания строки '404' в файле

3. на основании файла 404.html сделайте файл 403.html заменив в его содержимом все упоминания 404 на 403
	
	cd ../www/html #Выйти из папки logs и перейти в папку html
	cat 404.html > 403.html #Cоздать файл 403.html и скопировать содержимое файла 404.html
	sed -i 's/404/403/g' 403.html #Заменить '404' на '403')

4. модифицируйте права доступа и владельца файла index.html следующим образом: Текущий пользователь должен иметь
   права на чтение/запись. Группа www-data  на чтение, все остальные не имеют доступ к файлу
	
	sudo chmod 640 index.html #Изменение прав доступа к файлу index.html. Теперь текущий пользователь имеет 
				   права на чтение/запись, группа - на чтение, все остальные - не имеют доступ 
				   к файлу

5.сделайте символическую ссылку на файл access.log в папке  html

	sudo ln -s /var/logs/access.log mylink #Создание символической ссылки mylink на файл access.log в папке html

6. выведите на экран все файлы в папке var 
	ls -a /var #Выводит все файлы с папки var


7.  создайте новый файл 503.html  в папке html  со следующим содержимым
	<html>	
		<head>	
			<title>503</title>
		</head>
	</html>

	touch 503.html #Cоздает файл 503.html
	#Добавление указанной строки в файл 503.html
	sudo echo -e "<html>\n <head>\n  <title>503</title>\n  </head>\n</title>" > 503.html
